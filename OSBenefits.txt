%------------------------------------------------------------------------------
% Types
%------------------------------------------------------------------------------
tff(os_type,type,         os: $tType ).
tff(origin_type,type,     origin: $tType ).
tff(computer_type,type,   computer: $tType ).

%------------------------------------------------------------------------------
% Constants
%------------------------------------------------------------------------------
tff(linux_type,type,      linux: os ).
tff(windows_type,type,    windows: os ).
tff(osx_type,type,        osx: os ).
tff(unix_type,type,       unix: origin ).
tff(dos_type,type,        dos: origin ).
tff(my_computer_type,type,my_computer: computer ).

%------------------------------------------------------------------------------
% Function: origin_of maps an OS to its origin
%------------------------------------------------------------------------------
tff(origin_of_type,type,  origin_of: os > origin ).

%------------------------------------------------------------------------------
% Predicates
%------------------------------------------------------------------------------
tff(runs_type,type,       runs: ( computer * os ) > $o ).
tff(benefits_type,type,   benefits: ( computer * origin ) > $o ).

%------------------------------------------------------------------------------
% Axioms
%------------------------------------------------------------------------------

% Linux, Windows, and OSX are the only operating systems
tff(all_os,axiom,
    ! [O: os] : ( O = linux | O = windows | O = osx ) ).

% They are all different
tff(os_different_1,axiom, linux != windows ).
tff(os_different_2,axiom, linux != osx ).
tff(os_different_3,axiom, windows != osx ).

% The only operating system origins are UNIX and DOS
tff(all_origins,axiom,
    ! [R: origin] : ( R = unix | R = dos ) ).

% UNIX and DOS are different
tff(origins_different,axiom, unix != dos ).

% The origin of Linux and OSX is UNIX
tff(origin_linux,axiom, origin_of(linux) = unix ).
tff(origin_osx,axiom,   origin_of(osx) = unix ).

% The origin of Windows is DOS
tff(origin_windows,axiom, origin_of(windows) = dos ).

% All computers run one or more operating systems
tff(computers_run_os,axiom,
    ! [C: computer] : ? [O: os] : runs(C,O) ).

% A computer benefits from the origins of the operating systems that it runs
tff(benefits_from_origin,axiom,
    ! [C: computer, O: os] : ( runs(C,O) => benefits(C,origin_of(O)) ) ).

% My computer runs OSX
tff(my_computer_runs_osx,axiom, runs(my_computer,osx) ).

%------------------------------------------------------------------------------
% Conjecture
%------------------------------------------------------------------------------

% Therefore my computer benefits from UNIX
tff(my_computer_benefits_unix,conjecture, benefits(my_computer,unix) ).
